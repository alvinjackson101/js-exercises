Find the area of an triangle.

function myFunction(){ 
    var side1 = 15; 
    var side2 = 15; 
    var side3 = 15; 
    var s = (side1 + side2 + side3)/2;
    var area =  Math.sqrt(s*((s-side1)*(s-side2)*(s-side3)));
    console.log(area);
    document.getElementById("triarea").innerHTML = area;
}
